# --- Optional (Performance & Features) ---------------------------------------

# dxcam            - DirectX-based screen capture for Windows (fastest, recommended)
#                    pip install dxcam
# PyTurboJPEG      - Hardware-accelerated JPEG encoding (lower CPU usage)
#                    pip install PyTurboJPEG
# aiortc           - WebRTC support for screen streaming
#                    pip install aiortc
# cryptography     - Required for HTTPS/SSL certificate generation
#                    pip install cryptography
# python-uinput    - Virtual keyboard on Linux (requires root)
#                    pip install python-uinput

# --- External Tools ----------------------------------------------------------

# FFmpeg           - H264 hardware encoding (h264_amf / h264_nvenc / h264_qsv)
#                    Windows: winget install ffmpeg
#                    Linux:   sudo apt install ffmpeg
#                    macOS:   brew install ffmpeg
#                    *Without FFmpeg, streaming falls back to JPEG mode.

# Linux extra:     sudo apt install xdotool xclip xsel libportaudio2
